=============main.tf
# Specify the Docker host 
provider "docker" {
  host = "unix:///var/run/docker.sock"
}

# Download the latest Centos image
resource "docker_image" "myimg" {
  name = "centos:latest"
}

# Start the Container
resource "docker_container" "mycontainer" {
  name  = "testng"
  image = "${docker_image.myimg.latest}"
  ports {
    internal = "80"
    external = "80"
  }
}
